### Note 2: Enhanced api-client.ts Error Messages

Replace the current generic error throw with a status-specific message map:

```typescript
function errorMessageForStatus(status: number, fallback: string): string {
  if (status === 401) return 'Your session expired — please sign in again'
  if (status === 403) return 'Access Denied — contact your administrator'
  if (status === 404) return 'Resource not found'
  if (status >= 500) return 'Server error — try again in a few moments'
  return fallback
}
```

For network errors (fetch throws before a response), catch the error and throw:
```typescript
} catch (e) {
  if (e instanceof ApiError) throw e
  throw new ApiError(0, 'Connection failed — check your internet connection')
}
```

### Note 4: Four UI States Pattern

Every page must follow this exact pattern:

```tsx
// State declarations
const [data, setData] = useState<T | null>(null)
const [loading, setLoading] = useState(true)
const [error, setError] = useState<string | null>(null)

// Fetch function (extracted so Retry button can call it)
const load = useCallback(async () => {
  setLoading(true)
  setError(null)
  try {
    const token = await getToken()
    if (!token) { setError('Your session expired — please sign in again'); setLoading(false); return }
    const result = await apiClient<T>('/api/v1/...', token)
    setData(result)
  } catch (e) {
    setError(e instanceof Error ? e.message : 'Something went wrong')
  } finally {
    setLoading(false)
  }
}, [getToken])

useEffect(() => { load() }, [load])

// Render states
if (loading) return <LoadingSkeleton />
if (error) return <ErrorState message={error} onRetry={load} />
if (!data || (Array.isArray(data) && data.length === 0)) return <EmptyState />
return <PageContent data={data} />
```
